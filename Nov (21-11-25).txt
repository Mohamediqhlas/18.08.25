abstract class Vehicle {
    abstract void start();
    abstract void speedCheck(int speed) throws Exception;
}

class Bike extends Vehicle {

    void start() {
        System.out.println("Bike Started");
    }

    void start(String mode) {
        System.out.println("Bike Started in Mode: " + mode);
    }

    void speedCheck(int speed) throws Exception {
        if (speed > 150) {
            throw new Exception("Bike Overspeeding");
        }
        System.out.println("Bike Speed Safe");
    }
}

class Car extends Vehicle {

    void start() {
        System.out.println("Car Started");
    }

    void start(String mode) {
        System.out.println("Car Started in Mode: " + mode);
    }

    void speedCheck(int speed) throws Exception {
        if (speed > 200) {
            throw new Exception("Car Overspeeding");
        }
        System.out.println("Car Speed Safe");
    }
}

class Person {
    private String name;

    final void setName(String name) {
        this.name = name;
    }

    String getName() {
        return name;
    }
}
